## python中的几种排序方法

使用python来实现数据的几种排序方法，包括冒泡排序，直接选择排序，直接插入排序，快排，归并排序，二分法查找，堆排序等

#### 1. 冒泡排序
```python
def bubble_sort(list):
    for i in range(len(list)-1):
        current_status = False
        for j in range(len(list) - i -1):
            if list[j] > list[j + 1]:
                list[j], list[j + 1] = list[j + 1], list[j]
                current_status = True
        if not current_status:
            break
    return list
```

#### 2. 直接选择排序
```python
def select_sort(list):
    for i in range(len(list) - 1):
        min_index = i
        for j in range(i + 1, len(list)):
            if list[j] < list[min_index]:
                min_index = j
        list[i], list[min_index] = list[min_index], list[i]
    return list
```

#### 3. 直接插入排序
```python
def insert_sort(list):
    for i in range(1, len(list)):
        min_value = list[i]
        j = i - 1
        while j >= 0 and list[j] > min_value:
            list[j + 1] = list[j]
            j -= 1
        list[j + 1] = min_value
    return list
```

#### 4. 快速排序
```python
def partition(list, left, right):
    tmp = list[left]
    while left < right:
        while left < right and list[right] >= tmp:
            right -= 1
        list[left] = list[right]
        while left < right and list[left] <= tmp:
            left += 1
        list[right] = list[left]
    list[left] = tmp
    return list


# 快速排序
def quick_sort(list, left, right):
    if left < right:
        mid = partition(list, left, right)
        quick_sort(list, left, mid - 1)
        quick_sort(list, mid + 1, right)
        return list
```

#### 5. 归并排序
```python
# 归并排序
def merge(list, low, mid, high):
    """
    两段需要归并的序列从左到右遍历， 逐一比较，小的就放到tmp里
    :param list:
    :param low:
    :param mid:
    :param high:
    :return:
    """
    tmp = []
    i = low
    j = mid + 1
    while i <= mid and j <= high:
        if list[i] < list[j]:
            tmp.append(list[i])
            i += 1
        else:
            tmp.append(list[j])
            j += 1

    while i <= mid:
        tmp.append(list[i])
        i += 1

    while j <= high:
        tmp.append(list[j])
        j += 1
    list[low: high + 1] = tmp
    return list


def merge_sort(list, low, high):
    if low < high:
        mid = (low + high) // 2
        merge_sort(list, low, mid)
        merge_sort(list, mid + 1, high)
        return merge(list, low, mid, high)
```

#### 6. 二分法查找
```python
import math
# 二分法查找
def binary_sort(list, a):
    list.sort()
    low = 0
    high = len(list) - 1
    while low <= high:
        mid = math.floor((low + high) / 2)
        if list[mid] < a:
            low = mid + 1
        elif list[mid] > a:
            high = mid - 1
        else:
            return mid
    return None
```

#### 7. 堆排序
```python
# 堆排序
def sift(list, left, right):
    i = left
    j = 2 * i + 1
    tmp = list[i]
    while j <= right:
        if j < right and list[j] < list[j + 1]:
            j += 1
        if tmp < list[i]:
            list[i] = list[j]
            i = j
            j = 2 * j + 1
        else:
            break
    list[i] = tmp


def heap(list):
    n = len(list)
    for i in range(n//2 - 1, -1, -1):
        sift(list, i, n -1)
    for i in range(n -1, -1, -1):
        list[0], list[i] = list[i], list[0]
        sift(list, 0, i - 1)
```

#### 字符串全排序
```python
import itertools


def fun1(s=''):
    if len(s) <= 1:
        return [s]
    l1 = []
    for i in range(len(s)):
        for j in fun1(s[0:i] + s[i+1:]):
            l1.append(s[i] + j)
    return l1


def main():
    a = fun1('abc')
    print(len(a), a)


def func2():
    list1 = [1, 2, 3, 4]
    list2 = []
    for i in range(1, len(list1) + 1):
        iter = itertools.combinations(list1, i)
        list2.append(list(iter))
    print(list2)


if __name__ == '__main__':
    # main()
```
